
 create tabla usuarios(
   id 		INT PRIMARY KEY AUTO_INCREMENT,
   usuario 	VARCHAR(20),
   nombre 	VARCHAR(20),
   sexo 	VARCHAR(1),
   nivel 	TINYINT,
   email 	VARCHAR(50),
   telefono 	VARCHAR(20),
   marca 	VARCHAR(20),
   compania 	VARCHAR(20),
   saldo 	FLOAT,
   activo 	BOOLEAN
);
